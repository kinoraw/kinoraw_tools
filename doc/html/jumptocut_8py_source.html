399  newstrips.append(i)

401  #store meta channel
402  basechan = meta.channel
403  #look for upper and lower channels used by strips inside the meta
404  lowerchan = 32
405  upperchan = 0
406  for i in newstrips:
407  if i.channel < lowerchan: lowerchan = i.channel
408  if i.channel > upperchan: upperchan = i.channel
409  #calculate channel increment needed
410  deltachan = basechan - lowerchan
411  # reorder strips inside the meta

466  #print("removing: ",i.name, i.type)
467  for j in scn.sequence_editor.sequences_all: j.select = False

471

475  i.select = True
476  scn.sequence_editor.active_strip = i
477 
478  bpy.ops.sequencer.reload()
479

482  markers.remove(markers['IN'])
483  else:
484  markers["IN"].frame = original_in
485  if borrarout:
486  markers.remove(markers['OUT'])
487  else:
488  markers["OUT"].frame = original_out
489  scn.update()
490 
491 
492
